cmake_minimum_required(VERSION 3.17)

project(tritonexatrkxcpubackend LANGUAGES C CXX VERSION 0.1)

#
# Options
#
# Must include options required for this project as well as any
# projects included in this one by FetchContent.
#
# GPU support is disabled by default because exatrkxcpu backend
# doesn't use GPUs.
#
option(TRITON_ENABLE_GPU "Enable GPU support in backend" OFF)
option(TRITON_ENABLE_STATS "Include statistics collections in backend" ON)

# change "main" to "r22.02"
set(TRITON_COMMON_REPO_TAG "r22.02" CACHE STRING "Tag for triton-inference-server/common repo")
set(TRITON_CORE_REPO_TAG "r22.02" CACHE STRING "Tag for triton-inference-server/core repo")
set(TRITON_BACKEND_REPO_TAG "r22.02" CACHE STRING "Tag for triton-inference-server/backend repo")

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

#
# Dependencies
#
# FetchContent requires us to include the transitive closure of all
# repos that we depend on so that we can override the tags.
#
include(FetchContent)

FetchContent_Declare(
  repo-common
  GIT_REPOSITORY https://github.com/triton-inference-server/common.git
  GIT_TAG ${TRITON_COMMON_REPO_TAG}
  GIT_SHALLOW ON
)
FetchContent_Declare(
  repo-core
  GIT_REPOSITORY https://github.com/triton-inference-server/core.git
  GIT_TAG ${TRITON_CORE_REPO_TAG}
  GIT_SHALLOW ON
)
FetchContent_Declare(
  repo-backend
  GIT_REPOSITORY https://github.com/triton-inference-server/backend.git
  GIT_TAG ${TRITON_BACKEND_REPO_TAG}
  GIT_SHALLOW ON
)
FetchContent_MakeAvailable(repo-common repo-core repo-backend)

find_package(Torch REQUIRED)
#
# The backend must be built into a shared library. Use an ldscript to
# hide all symbols except for the TRITONBACKEND API.
#
configure_file(src/libtriton_exatrkxcpu.ldscript libtriton_exatrkxcpu.ldscript COPYONLY)

add_library(
  triton-exatrkxcpu-backend SHARED
  src/exatrkxcpu.cc
)

add_library(
  TritonExatrkxCPURuntimeBackend::triton-exatrkxcpu-backend ALIAS triton-exatrkxcpu-backend
)

target_compile_features(triton-exatrkxcpu-backend PRIVATE cxx_std_11)
target_compile_options(
  triton-exatrkxcpu-backend PRIVATE
  $<$<OR:$<CXX_COMPILER_ID:Clang>,$<CXX_COMPILER_ID:AppleClang>,$<CXX_COMPILER_ID:GNU>>:
    -Wall -Wextra -Wno-unused-parameter -Wno-type-limits -Werror>
  $<$<CXX_COMPILER_ID:MSVC>:/Wall /D_WIN32_WINNT=0x0A00 /EHsc>
)

target_link_libraries(
  triton-exatrkxcpu-backend
  PRIVATE
    ExaTrkXCPU
    ${TORCH_LIBRARIES}
    triton-core-serverapi   # from repo-core
    triton-core-backendapi  # from repo-core
    triton-core-serverstub  # from repo-core
    triton-backend-utils    # from repo-backend
)

if(WIN32)
  set_target_properties(
    triton-exatrkxcpu-backend PROPERTIES
    POSITION_INDEPENDENT_CODE ON
    OUTPUT_NAME triton_exatrkxcpu
  )
else()
  set_target_properties(
    triton-exatrkxcpu-backend PROPERTIES
    POSITION_INDEPENDENT_CODE ON
    OUTPUT_NAME triton_exatrkxcpu
    LINK_DEPENDS ${CMAKE_CURRENT_BINARY_DIR}/libtriton_exatrkxcpu.ldscript
    LINK_FLAGS "-Wl,--version-script libtriton_exatrkxcpu.ldscript"
  )
endif()

#
# Install
#
include(GNUInstallDirs)
set(INSTALL_CONFIGDIR ${CMAKE_INSTALL_LIBDIR}/cmake/TritonExatrkxCPUBackend)

install(
  TARGETS
    triton-exatrkxcpu-backend
  EXPORT
    triton-exatrkxcpu-backend-targets
  LIBRARY DESTINATION ${CMAKE_INSTALL_PREFIX}/backends/exatrkxcpu
  RUNTIME DESTINATION ${CMAKE_INSTALL_PREFIX}/backends/exatrkxcpu
)

install(
  EXPORT
    triton-exatrkxcpu-backend-targets
  FILE
    TritonExatrkxCPUBackendTargets.cmake
  NAMESPACE
    TritonExatrkxCPUBackend::
  DESTINATION
    ${INSTALL_CONFIGDIR}
)

include(CMakePackageConfigHelpers)
configure_package_config_file(
  ${CMAKE_CURRENT_LIST_DIR}/cmake/TritonExatrkxCPUBackendConfig.cmake.in
  ${CMAKE_CURRENT_BINARY_DIR}/TritonExatrkxCPUBackendConfig.cmake
  INSTALL_DESTINATION ${INSTALL_CONFIGDIR}
)

install(
  FILES
  ${CMAKE_CURRENT_BINARY_DIR}/TritonExatrkxCPUBackendConfig.cmake
  DESTINATION ${INSTALL_CONFIGDIR}
)

#
# Export from build tree
#
export(
  EXPORT triton-exatrkxcpu-backend-targets
  FILE ${CMAKE_CURRENT_BINARY_DIR}/TritonExatrkxCPUBackendTargets.cmake
  NAMESPACE TritonExatrkxCPUBackend::
)

export(PACKAGE TritonExatrkxCPUBackend)
